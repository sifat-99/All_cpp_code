    SELECT age

3.

    SELECT salary
    FROM CITIZEN

4.
SELECT * FROM CITIZEN
WHERE AGE < ANY (
    SELECT AGE
    FROM CITIZEN
    WHERE SALARY > 50000
);

5.
SELECT *,
CASE
WHEN SALARY>=60000 THEN 'HIGH INCOME'
WHEN SALARY BETWEEN 20000 AND 59999 THEN 'MEDIUM INCOME'
ELSE 'LOW INCOME'
END AS CATEGORY
FROM CITIZEN;

6.
SELECT C_NAME,
       CASE
           WHEN AGE < 30 THEN 'Youth'
           WHEN AGE BETWEEN 30 AND 60 THEN 'Adult'
           ELSE 'Senior'
       END AS Age_Group
FROM CITIZEN;


7.
SELECT *,
CASE
WHEN GENDER='Female' AND OCCUPATION!='Unemployed' THEN 'Working Women'
WHEN GENDER='Male' AND OCCUPATION!='Unemployed' THEN 'Man at working'
ELSE 'Loking for work'
END AS CATEGORY
FROM CITIZEN;


8.
SELECT C_NAME, C.GENDER, SALARY,
       CASE
           WHEN SALARY > (SELECT AVG(SALARY) FROM CITIZEN WHERE GENDER = C.GENDER) THEN 'Above Average'
           ELSE 'Below Average'
       END AS Salary_Status
FROM CITIZEN C;

9.
SELECT C_Name, SALARY FROM citizen
WHERE salary> (SELECT AVG(Salary) FROM citizen);
